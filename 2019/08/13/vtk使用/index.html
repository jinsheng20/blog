vtk 物体位置角度设置

vtkSmartPointer<vtkTransform> trans = vtkSmartPointer<vtkTransform>::New();
trans->PostMultiply();
trans->Translate(position->x, position->y, position->z);  //移动位置
trans->RotateX(position->roll);         //沿X轴旋转x度
trans->RotateY(position->pitch);         //沿Y轴旋转x度
trans->RotateZ(position->yaw);         //沿Z轴旋转x度

trans->Scale(0.5,0.5,0.5);  //实际实物缩放比例
actor->SetUserTransform(trans);

vtk相机位置

vtkSmartPointer<vtkCamera> camera = vtkSmartPointer<vtkCamera>::New();
camera->SetPosition(0, -1, 0);   //相机位置
camera->SetFocalPoint(0, 0, 0);  //焦点
camera->SetViewUp(0, 0, 1);      //相机方向
camera->Azimuth(150);
camera->Elevation(30);
camera->ComputeViewPlaneNormal(); //根据设置的相机位置、焦点等信息，重新计算视平面(View Plane)的法向量

pcl 点云数据平移 旋转

#include <vtkTransform.h>  
//两种实际改变点云位置的点云移动   注释部分没有验证 
//        //构造变化矩阵
//        Eigen::Matrix4f transform_1 = Eigen::Matrix4f::Identity();
//        float theta = M_PI/4;               //旋转的度数，这里是45度
//    //    transform_1 (0,0) = cos (theta);  //这里是绕的Z轴旋转
//    //    transform_1 (0,1) = -sin(theta);
//    //    transform_1 (1,0) = sin (theta);
//    //    transform_1 (1,1) = cos (theta);
//        //   transform_1 (0,2) = 0.3;   //这样会产生缩放效果
//        //   transform_1 (1,2) = 0.6;
//        //    transform_1 (2,2) = 1;
//        transform_1 (0,3) = 25; //这里沿X轴平移
//        transform_1 (1,3) = 30;
//        transform_1 (2,3) = 380;
Eigen::Affine3f transform_2 = Eigen::Affine3f::Identity();

// 在 X 轴上定义一个 2.5 米的平移.
transform_2.translation() << -5, 0.0, 0.0;
float theta = M_PI/2;
// 和前面一样的旋转; Z 轴上旋转 theta 弧度
transform_2.rotate (Eigen::AngleAxisf (theta, Eigen::Vector3f::UnitZ()));
// 执行变换，并将结果保存在新创建的 transformed_cloud 中
pcl::PointCloud<pcl::PointXYZ>::Ptr transformed_cloud (new pcl::PointCloud<pcl::PointXYZ> ());
// 可以使用 transform_1 或 transform_2; t它们是一样的
pcl::transformPointCloud (*basic_cloud_ptr, *transformed_cloud, transform_2);

读取 stl 文件并描绘

//常用到的头文件
#include <vtkRenderer.h>
#include <vtkImageViewer2.h>
#include <QVTKWidget.h>
#include <vtkJPEGReader.h>
#include <vtkTexture.h>
#include <vtkTextureMapToCylinder.h>
#include <vtkTransformTextureCoords.h>

#include <vtkImageActor.h>
#include <vtk3DSImporter.h>
#include <vtkActor.h>
#include <vtkActorCollection.h>
#include <vtkCamera.h>
#include <vtkNamedColors.h>
#include <vtkProperty.h>

#include <vtkSmartPointer.h>
#include <vtkPolyData.h>
#include <vtkSTLReader.h>

#include <vtkAxesActor.h>  //zuobiaoxi
#include <vtkLight.h>
#include <vtkTransform.h>  //zhuanhuang pingyi

std::string inputFilename = "~/luhu.stl";

vtkSmartPointer<vtkSTLReader> reader =
    vtkSmartPointer<vtkSTLReader>::New();
reader->SetFileName(inputFilename.c_str());
reader->Update();

vtkSmartPointer<vtkNamedColors> colors =
    vtkSmartPointer<vtkNamedColors>::New();
colors->SetColor("Bkg", 0.3, 0.4, 0.5);

vtkSmartPointer<vtkJPEGReader>bmpReader = vtkSmartPointer<vtkJPEGReader>::New();
bmpReader->SetFileName("/mnt/hgfs/linux_share/training_background.jpg");//读入纹理图

vtkSmartPointer<vtkTexture>texture = vtkSmartPointer<vtkTexture>::New();   //纹理映射
texture->SetInputConnection(bmpReader->GetOutputPort());
texture->InterpolateOn();


vtkSmartPointer<vtkTextureMapToCylinder> textureMap =
    vtkSmartPointer<vtkTextureMapToCylinder>::New();
textureMap->SetInputConnection(reader->GetOutputPort());

vtkSmartPointer<vtkTransformTextureCoords> transformTextureCoords =
    vtkSmartPointer<vtkTransformTextureCoords>::New();
transformTextureCoords->SetInputConnection(textureMap->GetOutputPort());
//transformTextureCoords->SetScale(1, 1, 0);


// Visualize
vtkSmartPointer<vtkPolyDataMapper> mapper =
    vtkSmartPointer<vtkPolyDataMapper>::New();
mapper->SetInputConnection(/*textureMap*/transformTextureCoords->GetOutputPort());

vtkSmartPointer<vtkActor> actor =
    vtkSmartPointer<vtkActor>::New();
actor->SetMapper(mapper);
actor->SetTexture(texture);//纹理为actor的一种属性

actor->GetProperty()->SetDiffuseColor(0.4, 0, 0);
actor->GetProperty()->SetDiffuse(0.6);    //漫反射
actor->GetProperty()->SetSpecular(0.4);
actor->GetProperty()->SetSpecularColor(
    colors->GetColor3d("White").GetData());
actor->GetProperty()->SetSpecularPower(30.0);

actor->GetProperty()->SetOpacity(1);
//actor->SetPosition(100, 0, 0);


vtkSmartPointer<vtkLight> myLight = vtkSmartPointer<vtkLight>::New(); //设置光照
myLight->SetColor(0,1,0);
myLight->SetPosition(0,0,6);
myLight->SetFocalPoint(renderer->GetActiveCamera()->GetFocalPoint());
renderer->AddLight(myLight);

renderer->AddActor(actor);
//renderer->AddActor(actor2);
renderer->ResetCamera();
//renderer->GetActiveCamera()->Dolly(0.1);  //显示缩放比例
renderer->ResetCameraClippingRange();
renderer->SetBackground(.3, .6, .3); // Background color green
